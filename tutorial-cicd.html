GitHub Actions Example

name: Performance Tests
on:
  push:
    branches: [main]
  pull_request:
    branches: [main]

jobs:
  load-test:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v3
      
      - name: Run LoadMagic.AI Test
        uses: loadmagicai/action@v1
        with:
          api-key: ${{ secrets.LOADMAGIC_API_KEY }}
          test-file: tests/loadtest.locust
          users: 100
          duration: 60s
          
      - name: Upload Results
        uses: actions/upload-artifact@v3
        if: always()
        with:
          name: load-test-results
          path: results/

Quality Gates

# Configure quality gates in loadmagic.yaml
quality_gates:
  - name: Response Time
    metric: response_time_p95
    threshold: 2000  # ms
    action: fail
    
  - name: Error Rate
    metric: error_rate
    threshold: 1  # percentage
    action: fail
    
  - name: Throughput
    metric: requests_per_second
    threshold: 100
    action: warn

Notification Setup

# Slack notification example
notifications:
  slack:
    webhook: ${{ secrets.SLACK_WEBHOOK }}
    channel: "#performance-tests"
    on_failure: true
    on_success: false
    
  email:
    to: team@example.com
    subject: "Load Test Results: ${{ github.ref }}"
    attach_results: true
