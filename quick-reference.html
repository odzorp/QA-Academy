Basic Locust Script

from locust import HttpUser, task, between class APITestUser(HttpUser): wait_time = between(1, 3) def on_start(self): # Login once per user self.client.post("/api/login", json={"user": "test"}) @task(3) def get_data(self): self.client.get("/api/data") @task(1) def post_action(self): self.client.post("/api/action", json={"action": "test"})
